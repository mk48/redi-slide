```javascript
let zooAnimals = [
  { type: "bird", name: "peacock" },
  { type: "bird", name: "parrot" },
  { type: "animal", name: "tiger" },
  { type: "civilization", name: "viking" },
  { type: "animal", name: "elephant" },
  { type: "bird", name: "crow" },
];

let birds = zooAnimals.filter((animal) => animal.type === "bird");
console.log(birds);
// [
//  { type: "bird", name: "peacock" },
//  { type: "bird", name: "parrot" },
//  { type: "bird", name: "crow" },
// ]
```
