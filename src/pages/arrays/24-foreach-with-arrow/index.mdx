# 1 - simple forEach

```javascript
let numbers = [1, 2, 3, 4];

let execute = function(n) {
  console.log(n);
}

numbers.forEach(execute);
```   

---

# 2 - forEach with arrow

```javascript
let numbers = [1, 2, 3, 4];

let execute = (n) => {
  console.log(n);
}

numbers.forEach(execute);
```   

---

# 3 - forEach with arrow inside

```javascript
let numbers = [1, 2, 3, 4];

numbers.forEach((n) => {
  console.log(n);
});
```   

---

# 4 - forEach with arrow inside - one line

```javascript
let numbers = [1, 2, 3, 4];

numbers.forEach((n) => console.log(n));
```
